\section*{Aufgabe 1.1}

Die \textbf{Konkatenation} zweier Sprachen ist $L_A \circ L_B = \{uv \mid u \in L_A,\, v \in L_B\}$.

\textbf{Beispiel:} $A = \{ab, c\}$, $B = \{1, 23\}$ $\Rightarrow$ $A \circ B = \{ab1,\; ab23,\; c1,\; c23\}$.

\bigskip
Seien $L_1 = \{w1 \mid w \in \{0,1\}^*\}$ und $L_2 = \{w0 \mid w \in \{0,1\}^*\}$.

\subsection*{1. $L_a = L_1 \cup L_2 = \{0,1\}^+$}

\begin{center}
\begin{tikzpicture}
    \node[state, initial] (q0) {$q_0$};
    \node[state, accepting, right=of q0] (q1) {$q_1$};
    \path (q0) edge node[above] {$0,1$} (q1)
          (q1) edge[loop above] node {$0,1$} ();
\end{tikzpicture}
\end{center}

\subsection*{2. $L_b = L_1 \cap L_2 = \emptyset$}

\begin{center}
\begin{tikzpicture}
    \node[state, initial] (q0) {$q_0$};
    \path (q0) edge[loop above] node {$0,1$} ();
\end{tikzpicture}
\end{center}

\subsection*{3. $L_c = L_1 \circ L_2 = \{w \in \{0,1\}^* \mid |w| \geq 2,\; w \text{ endet mit } 0\}$}

\begin{center}
\begin{tikzpicture}
    \node[state, initial] (q0) {$q_0$};
    \node[state, right=of q0] (q1) {$q_1$};
    \node[state, accepting, right=of q1] (q2) {$q_2$};
    \path (q0) edge node[above] {$0,1$} (q1)
          (q1) edge[bend left] node[above] {$0$} (q2)
          (q1) edge[loop above] node {$1$} ()
          (q2) edge[loop above] node {$0$} ()
          (q2) edge[bend left] node[below] {$1$} (q1);
\end{tikzpicture}
\end{center}
